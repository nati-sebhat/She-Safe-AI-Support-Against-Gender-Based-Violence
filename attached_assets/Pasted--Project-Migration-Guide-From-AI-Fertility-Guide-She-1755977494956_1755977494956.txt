2️⃣ Assistant Logic – assistant.py
This file controls the AI chatbot’s logic and personality.
✅ What to Do:
•	Open assistant.py
•	Replace all fertility-related code with this GBV-focused version:
# assistant.py – She Safe GBV AI Assistant
import openai, os
from dotenv import load_dotenv

load_dotenv()
openai.api_key = os.getenv("OPENAI_API_KEY")

SYSTEM_PROMPT = """
You are She Safe, an AI chatbot that helps users facing Gender-Based Violence (GBV).
Your role is to:
- Assess the situation safely and anonymously.
- Identify the type of GBV: physical, emotional, sexual, digital, economic, or other.
- Provide actionable guidance: emergency contacts, shelters, hotlines, legal aid.
- Suggest safety plans and coping strategies.
- Share awareness tips: consent, rights, healthy relationships.
- Offer mental health support: breathing exercises, survivor stories.
- Always be empathetic, concise, and privacy-focused.
"""

def get_response(user_input):
    response = openai.ChatCompletion.create(
        model="gpt-4o-mini",  # or gpt-5-nano
        messages=[
            {"role": "system", "content": SYSTEM_PROMPT},
            {"role": "user", "content": user_input}
        ]
    )
    return response['choices'][0]['message']['content']

# app.py – She Safe GBV Chatbot
import streamlit as st
from assistant import get_response
from questionnaire import QUESTIONNAIRE

# --- Page Settings ---
st.set_page_config(
    page_title="She Safe – GBV Support",
    page_icon="assets/logo.png",
    layout="centered"
)

# --- Custom CSS ---
st.markdown("""
    <style>
    body {
        font-family: 'Open Sans', sans-serif;
        background-color: #faf9f7;
        color: #333;
    }
    h1, h2, h3 {
        font-family: 'Montserrat', sans-serif;
        font-weight: 700;
        color: #222;
    }
    .stButton>button {
        background-color: #e07a5f;
        color: white;
        border-radius: 12px;
        padding: 0.6em 1.2em;
        font-size: 16px;
    }
    .panic-button>button {
        background-color: #c1121f;
        color: white;
        font-weight: bold;
    }
    </style>
""", unsafe_allow_html=True)

# --- Header ---
st.image("assets/logo.png", width=100)
st.title("🌍 She Safe – AI Chatbot Against GBV")
st.write("Confidential, safe, and supportive space for survivors and allies.")

# --- Panic Button ---
if st.button("🚨 Quick Exit", key="panic", help="Leave immediately"):
    st.markdown('<meta http-equiv="refresh" content="0; url=https://google.com" />', unsafe_allow_html=True)

# --- Emergency Support Section ---
st.subheader("🚨 Get Emergency Support")
st.write("If you are in immediate danger, please contact local authorities or hospitals:")
st.markdown("""
**Ethiopia**
- Police: 991
- Ambulance: 907
- Women & Children Hotline: 978  

**Cameroon**
- Police: 117
- Ambulance: 112  
- Gender Violence Hotline: 1521  
""")

# --- Questionnaire ---
with st.expander("📋 Start GBV Questionnaire"):
    for section, questions in QUESTIONNAIRE.items():
        st.subheader(section)
        for q in questions:
            st.text_input(q, key=q)

# --- Chatbot ---
st.subheader("💬 Talk to She Safe")
user_input = st.text_area("Type your message here...")
if st.button("Send"):
    if user_input.strip():
        with st.spinner("She Safe is responding..."):
            response = get_response(user_input)
        st.markdown(f"**She Safe:** {response}")
